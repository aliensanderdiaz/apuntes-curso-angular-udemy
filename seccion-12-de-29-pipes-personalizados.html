          // pipes/toggle-case.pipe.ts

          import { Pipe, PipeTransform } from '@angular/core';

          @Pipe({
            name: 'toggleCase',
          })
          export class ToggleCasePipe implements PipeTransform {
            transform(value: string): string {
              return value.toUpperCase();
            }
          }
          
          // products.module.ts

          declarations: [
            ...
            ToggleCasePipe
          ],

          // order.component.html

          {{ 'hola' | toggleCase }}
